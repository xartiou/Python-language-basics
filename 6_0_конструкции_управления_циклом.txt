	print('Верно')
		
		name = None

2. CONTINUE позволяет переходить на следующую итерацию и не выполнять команды
	number = 0
	n = int(input('Введите число n: '))
	while number <= n:
	    if number % 2 == 0:
	        number = number + 1
		continue
	print(number)#выведет нечетные числа
